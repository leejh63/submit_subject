static void apply_led_by_range(uint8_t range)
{
    switch (range)
    {
        case RANGE_OPEN:  led_blue();    break;
        case RANGE_STUCK: led_yellow();  break;
        case RANGE_CLOSE: led_red();     break;
        default:          led_all_off(); break;
    }
}

static volatile bool g_uart_tx_busy = false;
static volatile uint16_t g_uart_tx_inflight = 0u; // ŔüĽŰ Áß ±ćŔĚ

static inline uint16_t rb_next_u16(uint16_t x, uint16_t mod)
{
    return (uint16_t)((x + 1u) % mod);
}

// RX ring
static void uart_rx_ring_push(uint8_t b)
{
    uint16_t nw = rb_next_u16(g_uart_rx_w, UART_RX_RING_SIZE);
    if (nw == g_uart_rx_r) return; // overflow drop
    g_uart_rx_ring[g_uart_rx_w] = b;
    g_uart_rx_w = nw;
}

static int uart_rx_ring_pop(uint8_t *out)
{
    if (g_uart_rx_r == g_uart_rx_w) return 0;
    *out = g_uart_rx_ring[g_uart_rx_r];
    g_uart_rx_r = rb_next_u16(g_uart_rx_r, UART_RX_RING_SIZE);
    return 1;
}

// TX ring
static void uart_tx_ring_push(uint8_t b)
{
    uint16_t nw = rb_next_u16(g_uart_tx_w, UART_TX_RING_SIZE);
    if (nw == g_uart_tx_r) return; // overflow drop
    g_uart_tx_ring[g_uart_tx_w] = b;
    g_uart_tx_w = nw;
}

static uint16_t uart_tx_ring_count(void)
{
    if (g_uart_tx_w >= g_uart_tx_r) return (uint16_t)(g_uart_tx_w - g_uart_tx_r);
    return (uint16_t)(UART_TX_RING_SIZE - (g_uart_tx_r - g_uart_tx_w));
}

static uint16_t uart_tx_ring_contig_count(void)
{
    if (g_uart_tx_w >= g_uart_tx_r) return (uint16_t)(g_uart_tx_w - g_uart_tx_r);
    return (uint16_t)(UART_TX_RING_SIZE - g_uart_tx_r);
}

// TX API (non-blocking enqueue)
static void uart_write_char_nb(char c)
{
    uart_tx_ring_push((uint8_t)c);
}

static void uart_write_string_nb(const char *s)
{
    while (*s) uart_write_char_nb(*s++);
}

static void uart_tx_poll_complete(void)
{
    if (!g_uart_tx_busy) return;

    uint32_t rem = 0u;
    status_t st = LPUART_DRV_GetTransmitStatus(UART_INST, &rem);

    if (st == STATUS_SUCCESS)
    {
        // żĎ·á ˝ĂÁˇżˇ¸¸ r ŔĚµż
        g_uart_tx_r = (uint16_t)((g_uart_tx_r + g_uart_tx_inflight) % UART_TX_RING_SIZE);
        g_uart_tx_inflight = 0u;
        g_uart_tx_busy = false;
    }
    else if (st != STATUS_BUSY)
    {
        // żˇ·Ż/Áß´Ü: ´ÜĽř µĺ·Ó
        g_uart_tx_r = (uint16_t)((g_uart_tx_r + g_uart_tx_inflight) % UART_TX_RING_SIZE);
        g_uart_tx_inflight = 0u;
        g_uart_tx_busy = false;
    }
}

static void uart_tx_drain_step(void)
{
    if (g_uart_tx_busy) return;

    uint16_t cnt = uart_tx_ring_count();
    if (cnt == 0u) return;

    uint32_t rem = 0u;
    if (LPUART_DRV_GetTransmitStatus(UART_INST, &rem) == STATUS_BUSY)
    {
        g_uart_tx_busy = true;
        return;
    }

    uint16_t contig = uart_tx_ring_contig_count();
    uint16_t sendLen = contig;
    if (sendLen > 64u) sendLen = 64u;

    status_t st = LPUART_DRV_SendData(UART_INST, (uint8_t *)&g_uart_tx_ring[g_uart_tx_r], sendLen);
    if (st == STATUS_SUCCESS)
    {
        g_uart_tx_inflight = sendLen;
        g_uart_tx_busy = true;
    }
}

static void uart_rx_arm_chunk(void)
{
    if (g_uart_rx_armed) return;

    status_t st = LPUART_DRV_ReceiveData(UART_INST, g_uart_rx_chunk, UART_RX_CHUNK_SIZE);
    if (st == STATUS_SUCCESS)
        g_uart_rx_armed = true;
}

static void uart_rx_collect_and_rearm(void)
{
    if (!g_uart_rx_armed)
    {
        uart_rx_arm_chunk();
        return;
    }

    uint32_t rem = 0u;
    status_t st = LPUART_DRV_GetReceiveStatus(UART_INST, &rem);

    if (st == STATUS_SUCCESS)
    {
        for (uint32_t i = 0; i < UART_RX_CHUNK_SIZE; i++)
            uart_rx_ring_push(g_uart_rx_chunk[i]);

        g_uart_rx_armed = false;
        uart_rx_arm_chunk();
    }
    else if (st == STATUS_BUSY)
    {
        // still receiving...
    }
    else
    {
        (void)LPUART_DRV_AbortReceivingData(UART_INST);
        g_uart_rx_armed = false;
        uart_rx_arm_chunk();
    }
}

    (void)LPUART_DRV_Init(UART_INST, &lpUartState1, &lpuart_1_InitConfig0);

    g_uart_rx_w = g_uart_rx_r = 0u;
    g_uart_tx_w = g_uart_tx_r = 0u;

    g_uart_tx_busy = false;
    g_uart_tx_inflight = 0u;

    g_uart_rx_armed = false;
    uart_rx_arm_chunk();
}

// ============================================================
// cmd_feed_char: UI ŔÔ·Â ¶óŔÎżˇĽ­¸¸ µżŔŰ (CR/LF´Â ˝ÇÇŕ Ć®¸®°Ĺ)
// ============================================================
static void cmd_feed_char(char ch,
                          uint8_t *req_help,
                          uint8_t *req_status,
                          uint8_t *req_open,
                          uint8_t *req_close,
                          uint8_t *req_stuck)
{
    // backspace/delete
    if (ch == '\b' || ch == 0x7F)
    {
        if (g_cmd_len > 0u)
        {
            g_cmd_len--;
            g_cmd_buf[g_cmd_len] = '\0';

            // ĹÍąĚłÎżˇĽ­ ÇŃ ±ŰŔÚ Áöżě±â (ÄżĽ­°ˇ ŔÔ·Â ¶óŔÎżˇ ŔÖ´Ů´Â °ˇÁ¤)
            uart_write_string_nb("\b \b");
        }
        return;
    }

    // endline: command commit
    if (ch == '\r' || ch == '\n')
    {
        g_cmd_buf[g_cmd_len] = '\0';

        if (strcmp(g_cmd_buf, "help") == 0 || strcmp(g_cmd_buf, "h") == 0)
            *req_help = 1u;
        else if (strcmp(g_cmd_buf, "status") == 0 || strcmp(g_cmd_buf, "s") == 0)
            *req_status = 1u;
        else if (strcmp(g_cmd_buf, "open") == 0 || strcmp(g_cmd_buf, "o") == 0)
            *req_open = 1u;
        else if (strcmp(g_cmd_buf, "close") == 0 || strcmp(g_cmd_buf, "c") == 0)
            *req_close = 1u;
        else if (strcmp(g_cmd_buf, "stuck") == 0 || strcmp(g_cmd_buf, "st") == 0)
            *req_stuck = 1u;
        else if (g_cmd_buf[0] != '\0')
            ui_set_msg("unknown cmd (help/h)");

        // reset input line UI
        cmd_reset();
        ui_clear_input_line();
        return;
    }

    // normalize case
    if (ch >= 'A' && ch <= 'Z') ch = (char)(ch - 'A' + 'a');

    // printable only
    if (ch < 0x20 || ch > 0x7E) return;

    // append
    if (g_cmd_len < (CMD_BUF_SIZE - 1u))
    {
        g_cmd_buf[g_cmd_len++] = ch;
        g_cmd_buf[g_cmd_len] = '\0';

        // echo on input line
        uart_write_char_nb(ch);
    }
}

    { .id = FRAME_SLAVE_STATUS, .dir = LIN_DIR_RX, .buf = g_rxStatus, .len = DATA_SIZE, .timeoutCnt = TIMEOUT_LIN_CNT }

typedef struct
{
    uint8_t   frameIndex;
    uint32_t  periodTicks;   // 0ŔĚ¸é şńÁÖ±â
    uint32_t  nextDueTick;
    uint8_t   priority;      // ŔŰŔ»Ľö·Ď łôŔ˝
    bool      enabled;
} lin_sched_entry_t;

// 200ms = 200000us / 500us = 400 ticks
#define TICKS_200MS (400U)

enum
{
    SCHED_STATUS_POLL = 0,
    SCHED_COUNT
};

static lin_sched_entry_t g_sched[SCHED_COUNT] =
{
    { .frameIndex = FRAME_IDX_STATUS, .periodTicks = TICKS_200MS, .nextDueTick = 0U, .priority = 10U, .enabled = true }
};

static void scheduler_init(uint32_t nowTick)
{
    for (uint32_t i = 0; i < SCHED_COUNT; i++)
    {
        if (!g_sched[i].enabled) continue;
        g_sched[i].nextDueTick = nowTick + g_sched[i].periodTicks;
    }
}

static int32_t scheduler_pick_due(uint32_t nowTick)
{
    int32_t best = -1;
    uint8_t bestPrio = 0xFFU;

    for (uint32_t i = 0; i < SCHED_COUNT; i++)
    {
        if (!g_sched[i].enabled) continue;
        if (g_sched[i].periodTicks == 0U) continue;

        if ((int32_t)(nowTick - g_sched[i].nextDueTick) >= 0)
        {
            if (g_sched[i].priority < bestPrio)
            {
                bestPrio = g_sched[i].priority;
                best = (int32_t)i;
            }
        }
    }
    return best;
}

static void scheduler_on_fired(uint32_t schedIndex, uint32_t nowTick)
{
    if (g_sched[schedIndex].periodTicks != 0U)
        g_sched[schedIndex].nextDueTick = nowTick + g_sched[schedIndex].periodTicks;
}

// callback flags
#define LIN_F_PID_OK   (1UL << 0)
#define LIN_F_TX_DONE  (1UL << 1)
#define LIN_F_RX_DONE  (1UL << 2)
#define LIN_F_ERROR    (1UL << 3)
#define LIN_F_TIMEOUT  (1UL << 4)
static volatile uint32_t g_linFlags = 0U;

// ============================================================
// LIN Callback: flags only
// ============================================================
lin_callback_t CallbackHandler(uint32_t instance, lin_state_t *linState)
{
    lin_callback_t ret = linState->Callback;
    (void)instance;

    if (linState->timeoutCounterFlag)
    {
        linState->timeoutCounterFlag = false;
        g_linFlags |= (LIN_F_TIMEOUT | LIN_F_ERROR);
        g_linErrCode = 5;
        return ret;
    }

    g_lastEvent = (uint8_t)linState->currentEventId;

    switch (linState->currentEventId)
    {
        case LIN_PID_OK:        g_linFlags |= LIN_F_PID_OK;  break;
        case LIN_TX_COMPLETED:  g_linFlags |= LIN_F_TX_DONE; break;
        case LIN_RX_COMPLETED:  g_linFlags |= LIN_F_RX_DONE; break;

        case LIN_PID_ERROR:        g_linFlags |= LIN_F_ERROR; g_linErrCode = 1; break;
        case LIN_CHECKSUM_ERROR:   g_linFlags |= LIN_F_ERROR; g_linErrCode = 2; break;
        case LIN_READBACK_ERROR:   g_linFlags |= LIN_F_ERROR; g_linErrCode = 3; break;
        case LIN_FRAME_ERROR:      g_linFlags |= LIN_F_ERROR; g_linErrCode = 4; break;

        default: break;
    }

    return ret;
}

static void engine_reset_to_idle(void)
{
    (void)LIN_DRV_GotoIdleState(INST_LIN2);
    g_engState = ENG_IDLE;

    ENTER_CRITICAL();
    g_linFlags = 0U;
    EXIT_CRITICAL();
}

static bool engine_start_request(const lin_req_t *req)
{
    if (g_engState != ENG_IDLE) return false;

    // transaction start: clear errcode (success path will leave NONE unless error)
    g_linErrCode = 0;
    g_lastEvent  = 0;

    g_activeReq = *req;
    g_activeFrameIndex = req->frameIndex;

    (void)LIN_DRV_MasterSendHeader(INST_LIN2, g_frames[g_activeFrameIndex].id);
    g_engState = ENG_WAIT_PID;
    return true;
}

// żĎ·á ČÄĂł¸®: UI ąÝżµ (·Î±× ľřŔ˝)
static void on_request_done(const lin_req_t *req, bool ok)
{
    if (!ok)
    {
        g_ui_last_fail_tick = (uint32_t)g_tick500us;
        ui_set_msg("LIN FAIL");
        led_yellow();
        return;
    }

    g_ui_last_ok_tick = (uint32_t)g_tick500us;

    if (req->type == REQ_TYPE_STATUS)
    {
        g_ui_pos   = g_rxStatus[0];
        g_ui_range = g_rxStatus[1];
        g_ui_err   = g_rxStatus[2];
        g_ui_ctrl  = g_rxStatus[3];
        g_ui_slave_valid = true;

        apply_led_by_range(g_ui_range);
        ui_set_msg("STATUS updated");
        return;
    }

    if (req->type == REQ_TYPE_CMD)
    {
        // CMD ČÄ 1Č¸ STATUS µż±âČ­
        g_userStatusPending = true;
        ui_set_msg("CMD ok -> sync status");
        return;
    }
}

// żŁÁř step: callback flag ˝şłŔĽ¦ ČÄ FSM ÁřÇŕ
static void engine_step(void)
{
    uint32_t flags;
    ENTER_CRITICAL();
    flags = g_linFlags;
    g_linFlags = 0U;
    EXIT_CRITICAL();

    // app-level deadline
    if (g_engState != ENG_IDLE && g_activeReq.deadlineTick != 0U)
    {
        uint32_t now = (uint32_t)g_tick500us;
        if ((int32_t)(now - g_activeReq.deadlineTick) >= 0)
        {
            ui_set_msg("DEADLINE");
            on_request_done((const lin_req_t *)&g_activeReq, false);
            engine_reset_to_idle();
            return;
        }
    }

    // error handling + retry
    if (flags & LIN_F_ERROR)
    {
        if (g_activeReq.retryCount < g_activeReq.maxRetry)
        {
            g_activeReq.retryCount++;
            ui_set_msg("retry...");
            engine_reset_to_idle();
            (void)engine_start_request((const lin_req_t *)&g_activeReq);
            return;
        }

        on_request_done((const lin_req_t *)&g_activeReq, false);
        engine_reset_to_idle();
        return;
    }

    // WAIT_PID -> send/recv
    if (g_engState == ENG_WAIT_PID)
    {
        if (flags & LIN_F_PID_OK)
        {
            LIN_DRV_SetTimeoutCounter(INST_LIN2, g_frames[g_activeFrameIndex].timeoutCnt);

            if (g_frames[g_activeFrameIndex].dir == LIN_DIR_TX)
            {
                (void)LIN_DRV_SendFrameData(INST_LIN2,
                                            g_frames[g_activeFrameIndex].buf,
                                            g_frames[g_activeFrameIndex].len);
                g_engState = ENG_WAIT_TX;
            }
            else
            {
                (void)LIN_DRV_ReceiveFrameData(INST_LIN2,
                                               g_frames[g_activeFrameIndex].buf,
                                               g_frames[g_activeFrameIndex].len);
                g_engState = ENG_WAIT_RX;
            }
        }
        return;
    }

    // WAIT_TX -> done
    if (g_engState == ENG_WAIT_TX)
    {
        if (flags & LIN_F_TX_DONE)
        {
            on_request_done((const lin_req_t *)&g_activeReq, true);
            engine_reset_to_idle();
        }
        return;
    }

    // WAIT_RX -> done
    if (g_engState == ENG_WAIT_RX)
    {
        if (flags & LIN_F_RX_DONE)
        {
            on_request_done((const lin_req_t *)&g_activeReq, true);
            engine_reset_to_idle();
        }
        return;
    }
}

// ============================================================
// Dispatcher: CMD(event) > STATUS(user) > STATUS(periodic)
// ============================================================
static bool dispatcher_make_next_request(lin_req_t *outReq, uint32_t nowTick, int32_t *outSchedIndex)
{
    *outSchedIndex = -1;
    memset(outReq, 0, sizeof(*outReq));

    if (g_cmdPending)
    {
        uint8_t tgt = g_cmdTarget;
        req_source_t src = g_cmdSrc;
        g_cmdPending = false;

        memset(g_txCmd, 0, sizeof(g_txCmd));
        g_txCmd[0] = tgt;

        outReq->type = REQ_TYPE_CMD;
        outReq->src  = src;
        outReq->frameIndex = FRAME_IDX_CMD;

        outReq->maxRetry = 1U;
        outReq->retryCount = 0U;
        outReq->deadlineTick = nowTick + 300U; // 150ms
        return true;
    }

    if (g_userStatusPending)
    {
        g_userStatusPending = false;

        outReq->type = REQ_TYPE_STATUS;
        outReq->src  = REQ_SRC_UART;
        outReq->frameIndex = FRAME_IDX_STATUS;

        outReq->maxRetry = 2U;
        outReq->retryCount = 0U;
        outReq->deadlineTick = nowTick + 400U; // 200ms
        return true;
    }

    int32_t si = scheduler_pick_due(nowTick);
    if (si >= 0)
    {
        *outSchedIndex = si;

        outReq->type = REQ_TYPE_STATUS;
        outReq->src  = REQ_SRC_PERIODIC;
        outReq->frameIndex = g_sched[si].frameIndex;

        outReq->maxRetry = 1U;
        outReq->retryCount = 0U;
        outReq->deadlineTick = nowTick + 400U;
        return true;
    }

    return false;
}

// ============================================================
// Input hooks
// ============================================================
static void request_motor_target(req_source_t src, uint8_t target)
{
    if (target > MOTOR_TARGET_STUCK) target = MOTOR_TARGET_STUCK;

    g_cmdSrc = src;
    g_cmdTarget = target;
    g_cmdPending = true; // last-wins
}

static void request_status_now(req_source_t src)
{
    (void)src;
    g_userStatusPending = true; // coalesce
}

// ============================================================
// UI refresh: overwrite fixed region, then restore cursor to input
// ============================================================
static void ui_refresh(void)
{
    char line[96];

    // [2] LIN STATE
    term_cursor_pos(UI_ROW_LIN_1, 1); term_clear_line();
    {
        const char *comm = (g_linErrCode == 0) ? "OK" : "ERR";
        (void)snprintf(line, sizeof(line), "comm : %s   lastErr=%s   lastEvent=%u   msg=%s",
                       comm, lin_err_str(g_linErrCode), (unsigned)g_lastEvent, g_ui_msg);
        uart_write_string_nb(line);
    }

    term_cursor_pos(UI_ROW_LIN_2, 1); term_clear_line();
    {
        const char *st =
            (g_engState == ENG_IDLE)     ? "IDLE" :
            (g_engState == ENG_WAIT_PID) ? "WAIT_PID" :
            (g_engState == ENG_WAIT_TX)  ? "WAIT_TX" : "WAIT_RX";

        (void)snprintf(line, sizeof(line), "state: %s   tick=%lu",
                       st, (unsigned long)g_tick500us);
        uart_write_string_nb(line);
    }

    term_cursor_pos(UI_ROW_LIN_3, 1); term_clear_line();
    {
        if (g_engState != ENG_IDLE)
        {
            const char *t = (g_activeReq.type == REQ_TYPE_CMD) ? "CMD" :
                            (g_activeReq.type == REQ_TYPE_STATUS) ? "STATUS" : "UNK";
            (void)snprintf(line, sizeof(line),
                           "req  : %s   retry=%u/%u   deadline=%lu",
                           t,
                           (unsigned)g_activeReq.retryCount,
                           (unsigned)g_activeReq.maxRetry,
                           (unsigned long)g_activeReq.deadlineTick);
        }
        else
        {
            uint32_t now = (uint32_t)g_tick500us;

            // g_tick500us´Â 0.5ms ´ÜŔ§ tick
            // ms = ticks / 2
            uint32_t ok_age_ms   = 0U;
            uint32_t fail_age_ms = 0U;

            if (g_ui_last_ok_tick != 0U)
                ok_age_ms = (uint32_t)((now - g_ui_last_ok_tick) / 2U);

            if (g_ui_last_fail_tick != 0U)
                fail_age_ms = (uint32_t)((now - g_ui_last_fail_tick) / 2U);

            (void)snprintf(line, sizeof(line),
                           "req  : (none)   lastOK=%lums ago   lastFAIL=%lums ago",
                           (unsigned long)ok_age_ms,
                           (unsigned long)fail_age_ms);
        }
        uart_write_string_nb(line);
    }

    // [3] SLAVE STATUS
    term_cursor_pos(UI_ROW_SLAVE_1, 1); term_clear_line();
    if (g_ui_slave_valid)
        (void)snprintf(line, sizeof(line), "pos  : %3u %%", (unsigned)g_ui_pos);
    else
        (void)snprintf(line, sizeof(line), "pos  : (no data)");
    uart_write_string_nb(line);

    term_cursor_pos(UI_ROW_SLAVE_2, 1); term_clear_line();
    if (g_ui_slave_valid)
        (void)snprintf(line, sizeof(line), "range: %s", range_str(g_ui_range));
    else
        (void)snprintf(line, sizeof(line), "range: (no data)");
    uart_write_string_nb(line);

    term_cursor_pos(UI_ROW_SLAVE_3, 1); term_clear_line();
    if (g_ui_slave_valid)
        (void)snprintf(line, sizeof(line), "ctrl : %s   err=0x%02X",
                       control_str(g_ui_ctrl), (unsigned)g_ui_err);
    else
        (void)snprintf(line, sizeof(line), "ctrl : (no data)");
    uart_write_string_nb(line);

    // Restore cursor to input line at current command length
    term_cursor_pos(UI_ROW_INPUT_LINE, (uint8_t)(UI_INPUT_COL_TEXT + g_cmd_len));
}

// ============================================================
// Master task (integrated)
// ============================================================
static void lin_master_task(void)
{
    uart_io_init();

    // LIN init
    (void)LIN_DRV_Init(INST_LIN2, &lin2_InitConfig0, &lin2_State);
    (void)LIN_DRV_InstallCallback(INST_LIN2, (lin_callback_t)CallbackHandler);

    led_all_off();

    // scheduler init
    scheduler_init((uint32_t)g_tick500us);
    engine_reset_to_idle();

    // UI init
    cmd_reset();
    ui_init();
    ui_set_msg("type: open/close/stuck/status/help");
    ui_refresh();

    uart_tx_drain_step();

    // ąöĆ° µđąŮżî˝ş(20ms)
    uint32_t lastBtnOpenTick  = 0U;
    uint32_t lastBtnCloseTick = 0U;
    const uint32_t debounceTicks = 40U; // 20ms

    uint32_t lastUi = (uint32_t)g_tick500us;

    for (;;)
    {
        // UART progress
        uart_rx_collect_and_rearm();
        uart_tx_poll_complete();
        uart_tx_drain_step();

        // engine step
        engine_step();

        // CLI parse
        uint8_t req_help = 0u, req_status = 0u, req_open = 0u, req_close = 0u, req_stuck = 0u;
        uint8_t b;

        while (uart_rx_ring_pop(&b))
        {
            cmd_feed_char((char)b, &req_help, &req_status, &req_open, &req_close, &req_stuck);
        }

        // handle commands (no logs -> UI msg¸¸ şŻ°ć)
        if (req_help)
        {
            ui_set_msg("cmd: open(o) close(c) stuck(st) status(s) help(h)");
        }

        if (req_open)
        {
            ui_set_msg("CMD: OPEN");
            request_motor_target(REQ_SRC_UART, MOTOR_TARGET_LEFT);
        }
        if (req_close)
        {
            ui_set_msg("CMD: CLOSE");
            request_motor_target(REQ_SRC_UART, MOTOR_TARGET_RIGHT);
        }
        if (req_stuck)
        {
            ui_set_msg("CMD: STUCK");
            request_motor_target(REQ_SRC_UART, MOTOR_TARGET_STUCK);
        }
        if (req_status)
        {
            ui_set_msg("STATUS: requested");
            request_status_now(REQ_SRC_UART);
        }

        // Buttons -> requests (with debounce)
        if (g_btnOpen)
        {
            g_btnOpen = false;
            uint32_t now = (uint32_t)g_tick500us;
            if ((uint32_t)(now - lastBtnOpenTick) >= debounceTicks)
            {
                lastBtnOpenTick = now;
                ui_set_msg("BTN: OPEN");
                request_motor_target(REQ_SRC_BTN, MOTOR_TARGET_LEFT);
            }
        }

        if (g_btnClose)
        {
            g_btnClose = false;
            uint32_t now = (uint32_t)g_tick500us;
            if ((uint32_t)(now - lastBtnCloseTick) >= debounceTicks)
            {
                lastBtnCloseTick = now;
                ui_set_msg("BTN: CLOSE");
                request_motor_target(REQ_SRC_BTN, MOTOR_TARGET_RIGHT);
            }
        }

        // dispatch next request if idle
        if (g_engState == ENG_IDLE)
        {
            lin_req_t req;
            int32_t schedIndex;
            uint32_t now = (uint32_t)g_tick500us;

            if (dispatcher_make_next_request(&req, now, &schedIndex))
            {
                if (engine_start_request(&req))
                {
                    if (schedIndex >= 0)
                        scheduler_on_fired((uint32_t)schedIndex, now);
                }
            }
        }

        // UI refresh periodic
        {
            uint32_t now = (uint32_t)g_tick500us;
            if ((uint32_t)(now - lastUi) >= UI_REFRESH_TICKS)
            {
                lastUi = now;
                ui_refresh();
            }
        }

        if (exit_code != 0) break;
    }

    // leaving: show cursor back
    term_show_cursor();
}

// ============================================================
// main
// ============================================================
int main(void)
{
    CLOCK_SYS_Init(g_clockManConfigsArr, CLOCK_MANAGER_CONFIG_CNT,
                   g_clockManCallbacksArr, CLOCK_MANAGER_CALLBACK_CNT);
    CLOCK_SYS_UpdateConfiguration(0U, CLOCK_MANAGER_POLICY_AGREEMENT);

    PINS_DRV_Init(NUM_OF_CONFIGURED_PINS0, g_pin_mux_InitConfigArr0);

#if USE_LIN_XCVR
    PINS_DRV_SetPinsDirection(LIN_XCVR_ENABLE_GPIO_PORT, LIN_XCVR_ENABLE_MASK);
    PINS_DRV_SetPins(LIN_XCVR_ENABLE_GPIO_PORT, LIN_XCVR_ENABLE_MASK);
#endif

    // Button IRQ (PORTC)
    INT_SYS_InstallHandler(BTN_PORT_IRQn, BTNPORT_IRQHandler, (isr_t *)NULL);
    INT_SYS_EnableIRQ(BTN_PORT_IRQn);

    // LPTMR
    LPTMR_DRV_Init(INST_LPTMR_1, &lptmr_1_config0, false);
    INT_SYS_InstallHandler(LPTMR0_IRQn, LPTMR_ISR, (isr_t *)NULL);
    INT_SYS_EnableIRQ(LPTMR0_IRQn);
    LPTMR_DRV_StartCounter(INST_LPTMR_1);

    lin_master_task();

    for (;;)
    {
        if (exit_code != 0) break;
    }
    return exit_code;
}
