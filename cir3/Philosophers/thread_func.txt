
#include <sys/time.h>   // gettimeofday
#include <unistd.h>     // usleep
#include <pthread.h>    // pthread_*
#include <semaphore.h>  // sem_*
#include <fcntl.h>      // O_CREAT 등

int main() {
    // 1. 시간 함수
    struct timeval tv;
    gettimeofday(&tv, NULL);            // 현재 시간 얻기
    long timestamp_ms = tv.tv_sec * 1000 + tv.tv_usec / 1000;
    usleep(100000);                     // 100ms 대기

    // 2. 스레드 생성 및 관리
    pthread_t thread;
    void *(*start_routine)(void *) = NULL; // 실제 함수 포인터 필요
    void *arg = NULL;                   // 실제 인자 필요
    pthread_create(&thread, NULL, start_routine, arg); // 스레드 생성
    // pthread_detach(thread);           // (옵션) detached 상태
    pthread_join(thread, NULL);         // 스레드 종료 대기

    // 3. 뮤텍스 사용
    pthread_mutex_t mutex;
    pthread_mutex_init(&mutex, NULL);   // 뮤텍스 초기화
    pthread_mutex_lock(&mutex);         // 뮤텍스 잠금
    // 임계 구역 (공유 자원 접근)
    pthread_mutex_unlock(&mutex);       // 뮤텍스 잠금 해제
    pthread_mutex_destroy(&mutex);      // 뮤텍스 파괴

    // 4. 세마포어 사용 (보너스)
    sem_t *sem = sem_open("/mysem", O_CREAT, 0666, 5); // 세마포어 생성/열기
    sem_wait(sem);                      // 세마포어 잠금
    // 임계 구역 (공유 자원 접근)
    sem_post(sem);                      // 세마포어 잠금 해제
    sem_close(sem);                     // 세마포어 닫기
    sem_unlink("/mysem");               // 세마포어 이름 제거

    return 0;
}





#include <stdio.h>
#include <sys/time.h>
#include <unistd.h>
#include <pthread.h>
#include <semaphore.h>

// 전역 리소스
pthread_mutex_t mtx;
sem_t *sem;
struct timeval start_time;

// 스레드 함수
void *worker(void *arg) {
    struct timeval now;
    // 시간 측정 예시
    gettimeofday(&now, NULL);
    printf("Time since start: %ld ms
",
        (now.tv_sec - start_time.tv_sec) * 1000
        + (now.tv_usec - start_time.tv_usec) / 1000);

    // 뮤텍스 잠금
    pthread_mutex_lock(&mtx);
    // 세마포어 wait
    sem_wait(sem);

    // 크리티컬 섹션
    printf("In critical section
");
    usleep(500 * 1000); // 500 ms

    // 세마포어 post
    sem_post(sem);
    // 뮤텍스 언락
    pthread_mutex_unlock(&mtx);

    return NULL;
}

int main(void) {
    pthread_t tid;

    // 1) 시간 기록
    gettimeofday(&start_time, NULL);

    // 2) 뮤텍스 초기화
    pthread_mutex_init(&mtx, NULL);

    // 3) 세마포어 초기화
    sem_unlink("/res");
    sem = sem_open("/res", O_CREAT | O_EXCL, 0644, 1);

    // 4) 스레드 생성
    pthread_create(&tid, NULL, worker, NULL);

    // 5) 스레드 종료 대기
    pthread_join(tid, NULL);

    // 6) 자원 정리
    pthread_mutex_destroy(&mtx);
    sem_close(sem);
    sem_unlink("/res");

    return 0;
}








gettimeofday (struct timeval *tv, struct timezone *tz)
설명: 현재 시각을 초(tv_sec)·마이크로초(tv_usec) 단위로 가져옴
반환값: 0(성공), –1(실패)

tv (struct timeval *, NULL 불가)

결과 저장용 구조체. 필드:

tv_sec (time_t): 초

tv_usec (suseconds_t): 마이크로초

tz (struct timezone *, NULL 허용)

타임존 정보(deprecated). 항상 NULL로 사용

usleep (useconds_t usec)
설명: 현재 스레드를 usec 마이크로초만큼 일시 중지
반환값: 0(성공), –1(실패)

usec (useconds_t, NULL 불가)

대기 시간(마이크로초). 0 이상

pthread_create (pthread_t *thread, const pthread_attr_t *attr, void *(*start_routine)(void *), void *arg)
설명: 새 스레드를 생성하고 start_routine(arg)를 실행
반환값: 0(성공), 오류코드(EINVAL, EAGAIN, EPERM 등)

thread (pthread_t *, NULL 불가)

생성된 스레드 ID 저장용

attr (const pthread_attr_t *, NULL 허용)

스레드 속성 지정(if NULL → 기본: JOINABLE, 기본 스택)

주요 설정 함수:

pthread_attr_setdetachstate(attr, PTHREAD_CREATE_DETACHED)

pthread_attr_setstacksize(attr, size_t)

pthread_attr_setguardsize(attr, size_t)

pthread_attr_setschedpolicy(attr, SCHED_FIFO/SCHED_RR/SCHED_OTHER)

pthread_attr_setinheritsched(attr, PTHREAD_EXPLICIT_SCHED/PTHREAD_INHERIT_SCHED)

start_routine (void *(*)(void *), NULL 불가)

스레드 진입 함수

arg (void *, NULL 허용)

진입 함수에 전달할 인자

pthread_join (pthread_t thread, void **retval)
설명: 지정 스레드 종료를 기다리고 반환값 수신
반환값: 0(성공), 오류코드(EINVAL, ESRCH, EDEADLK 등)

thread (pthread_t, NULL 불가)

대기할 스레드 ID

retval (void **, NULL 허용)

종료 함수의 리턴값 수신. NULL → 무시

pthread_detach (pthread_t thread)
설명: 스레드를 detached 상태로 설정(종료 시 자원 자동 해제)
반환값: 0(성공), 오류코드(EINVAL, ESRCH 등)

thread (pthread_t, NULL 불가)

detached로 설정할 스레드 ID

pthread_mutex_init (pthread_mutex_t *mutex, const pthread_mutexattr_t *attr)
설명: 뮤텍스 객체 초기화
반환값: 0(성공), 오류코드

mutex (pthread_mutex_t *, NULL 불가)

초기화 대상 뮤텍스

attr (const pthread_mutexattr_t *, NULL 허용)

뮤텍스 속성(if NULL → 기본: NORMAL, pshared=0)

주요 설정: pthread_mutexattr_settype(attr, PTHREAD_MUTEX_RECURSIVE/ERRORCHECK/NORMAL), pthread_mutexattr_setpshared(attr, PTHREAD_PROCESS_SHARED/PRIVATE)

pthread_mutex_lock (pthread_mutex_t *mutex)
설명: 뮤텍스 잠금(이미 잠겨 있으면 대기)
반환값: 0(성공), 오류코드

mutex (pthread_mutex_t *, NULL 불가)

pthread_mutex_unlock (pthread_mutex_t *mutex)
설명: 뮤텍스 해제
반환값: 0(성공), 오류코드

mutex (pthread_mutex_t *, NULL 불가)

pthread_mutex_destroy (pthread_mutex_t *mutex)
설명: 뮤텍스 파괴 (destroy 전 반드시 unlock 완료)
반환값: 0(성공), 오류코드

mutex (pthread_mutex_t *, NULL 불가)

sem_open (const char *name, int oflag, mode_t mode, unsigned int value)
설명: 명명된 세마포어 생성/열기
반환값: sem_t * (성공), SEM_FAILED (실패)

name (const char *, NULL 불가)

세마포어 이름, 반드시 ‘/’로 시작

oflag (int, NULL 불가)

O_CREAT (생성), O_EXCL (기존 존재 시 실패), 조합 가능

mode (mode_t, NULL 허용)

권한 비트(0644/0666 등), O_CREAT 시만 의미

value (unsigned int, NULL 허용)

초기 카운터 값

sem_wait (sem_t *sem)
설명: 세마포어 카운터 감소, 0 이하면 블록
반환값: 0(성공), –1(실패)

sem (sem_t *, NULL 불가)

sem_post (sem_t *sem)
설명: 세마포어 카운터 증가(언락)
반환값: 0(성공), –1(실패)

sem (sem_t *, NULL 불가)

sem_close (sem_t *sem)
설명: 현재 프로세스에서 세마포어 핸들 닫기
반환값: 0(성공), –1(실패)

sem (sem_t *, NULL 불가)

sem_unlink (const char *name)
설명: 이름 공간에서 세마포어 제거
반환값: 0(성공), –1(실패)

name (const char *, NULL 불가)

제거할 세마포어 이름, ‘/’로 시작





1. 시간 관련 자료형·구조체
자료형/구조체		정의 위치			설명
struct timeval	<sys/time.h>	초(tv_sec, time_t)와 마이크로초(tv_usec, suseconds_t) 단위의 시간 저장
struct timezone	<sys/time.h>	타임존 정보(사용 안 함, 대체로 NULL 전달)
time_t		<time.h>		POSIX에서 “초” 단위 시각을 저장하는 정수형
suseconds_t		<sys/types.h>	“마이크로초” 단위 시각을 저장하는 정수형
useconds_t		<unistd.h>		usleep()에 전달하는 마이크로초 단위 지연 시간 타입

2. 스레드 관련 자료형·구조체
자료형	정의 위치	설명
pthread_t		<pthread.h>		스레드 ID를 표현하는 불투명 타입
pthread_attr_t	<pthread.h>		스레드 속성 객체 타입 (스레드 생성 옵션 지정용)
void *(*)(void *)				스레드 시작 함수 포인터: void *func(void *arg) 형태
void *					스레드에 전달하거나 반환하는 데 쓰는 범용 포인터
int						에러 코드 및 단순 인자·반환용 정수형

3. 뮤텍스 관련 자료형·구조체
자료형				정의 위치		설명
pthread_mutex_t		<pthread.h>		뮤텍스(상호 배제 락) 객체 타입
pthread_mutexattr_t	<pthread.h>		뮤텍스 속성 객체 타입 (보통 NULL로 기본 속성 사용)

4. 세마포어 관련 자료형·구조체 (보너스 파트 전용)
자료형			정의 위치		설명
sem_t			<semaphore.h>	세마포어 객체 타입
mode_t		<sys/types.h>	파일 모드(권한) 지정용 정수형
unsigned int				세마포어 초기값(value) 지정용 부호 없는 정수형
sem_t *		<semaphore.h>	명명된 세마포어 핸들 — sem_open이 반환하는 포인터
