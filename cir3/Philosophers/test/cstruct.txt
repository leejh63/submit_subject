C 언어에서 enum, struct, union은 모두 사용자 정의 자료형입니다. 각각의 개념과 사용법을 간단한 예시와 함께 정리해드릴게요.

✅ 1. enum (열거형)
목적: 관련된 상수 값에 이름을 붙이기 위해 사용
기본값: 첫 값은 0, 이후 자동으로 1씩 증가

c
Copy
Edit
#include <stdio.h>

enum Color { RED, GREEN, BLUE };

int main() {
    enum Color c = GREEN;
    printf("Color value: %d\n", c);  // 출력: 1
    return 0;
}
직접 값 지정도 가능:

c
Copy
Edit
enum Weekday { MON = 1, TUE = 3, WED = 5 };  // 각각 1, 3, 5
✅ 2. struct (구조체)
목적: 서로 다른 타입의 데이터를 하나로 묶기 위해 사용

c
Copy
Edit
#include <stdio.h>

struct Point {
    int x;
    int y;
};

int main() {
    struct Point p1 = { .x = 3, .y = 5 };
    printf("Point: (%d, %d)\n", p1.x, p1.y);
    return 0;
}
typedef로 간단하게:

c
Copy
Edit

    int width;
    int height;
} Size;

Size s = {800, 600};
✅ 3. union (공용체)
목적: 여러 타입을 한 공간에 겹쳐서 저장할 때 사용 (메모리를 절약)

c
Copy
Edit
#include <stdio.h>

union Data {
    int i;
    float f;
    char str[4];
};

int main() {
    union Data d;
    d.i = 123;
    printf("int: %d\n", d.i);
    d.f = 3.14;
    printf("float: %f\n", d.f); // d.i는 덮어씌워짐
    return 0;
}
✅ union은 멤버 중 가장 큰 크기만큼만 메모리를 사용하며, 하나의 멤버만 유효

✅ 비교 정리
구분	목적	특징
enum	상수 그룹 정의	정수로 처리, 가독성 향상
struct	여러 멤버를 하나로 묶음	각 멤버가 별도의 메모리를 가짐
union	하나의 공간을 여러 타입으로 사용	모든 멤버가 같은 메모리를 공유

필요하다면 enum + struct + union을 함께 사용하는 예시도 만들어드릴 수 있어요. 더 복잡한 구조체 배열, 포인터 사용 예시가 궁금하면 말씀 주세요!

각 센서의 타입은 enum, 센서 데이터는 union, 그리고 하나의 센서 구조는 struct로 표현합니다.

✅ 전체 코드 예시
c
Copy
Edit
// sensor.h
#ifndef SENSOR_H
#define SENSOR_H

#include <stdio.h>

typedef enum {
    SENSOR_TEMPERATURE,
    SENSOR_HUMIDITY,
    SENSOR_ACCELERATION
} SensorType;

typedef union {
    float temperature;       // 온도 센서
    float humidity;          // 습도 센서
    struct {
        float x, y, z;       // 가속도 센서
    } accel;
} SensorData;

typedef struct {
    SensorType type;
    char name[32];
    SensorData data;
} Sensor;

void print_sensor(const Sensor *s);

#endif
c
Copy
Edit
// sensor.c
#include "sensor.h"

void print_sensor(const Sensor *s) {
    printf("Sensor: %s\n", s->name);
    switch (s->type) {
        case SENSOR_TEMPERATURE:
            printf("  Type: Temperature\n");
            printf("  Data: %.2f°C\n", s->data.temperature);
            break;
        case SENSOR_HUMIDITY:
            printf("  Type: Humidity\n");
            printf("  Data: %.2f%%\n", s->data.humidity);
            break;
        case SENSOR_ACCELERATION:
            printf("  Type: Acceleration\n");
            printf("  Data: x=%.2f, y=%.2f, z=%.2f\n",
                s->data.accel.x, s->data.accel.y, s->data.accel.z);
            break;
    }
}

네, 완전히 가능합니다.
C에서는 구조체 포인터에 malloc()으로 여러 개를 동적할당한 경우, struct_ptr[i] 방식으로 배열처럼 접근 가능합니다.

✅ 예제로 확인해보기
c
Copy
Edit
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    int id;
    char name[20];
} Person;

int main() {
    int count = 10;
    Person *people = malloc(sizeof(Person) * count);
    if (!people) {
        perror("malloc failed");
        return 1;
    }

    // 초기화
    for (int i = 0; i < count; i++) {
        people[i].id = i + 1;
        snprintf(people[i].name, sizeof(people[i].name), "Person_%d", i + 1);
    }

    // 접근
    for (int i = 0; i < count; i++) {
        printf("ID: %d, Name: %s\n", people[i].id, people[i].name);
    }

    free(people);
    return 0;
}
✅ 요점 정리
내용	설명
malloc(sizeof(T) * N)	구조체 T를 N개 연속으로 할당함 (배열처럼)
ptr[i]	내부적으로 *(ptr + i)와 같음, 배열처럼 접근 가능
구조체 멤버 접근	ptr[i].member 또는 (*(ptr + i)).member 모두 가능

🚨 주의할 점
malloc() 후 NULL 체크 필수 (메모리 부족 대비)

free()를 호출하지 않으면 메모리 누수 발생

배열처럼 접근하므로, i < count 조건 엄수

ChatGPT said:
C에서는 실행 중(runtime)에 변수의 자료형(type)을 직접적으로 알아내는 방법은 없습니다.
왜냐하면 C는 정적 타입(static type) 언어이기 때문에, 타입은 컴파일 타임에 이미 결정되고, 런타임에는 타입 정보가 남아있지 않기 때문입니다.

하지만 상황에 따라 아래와 같은 우회적인 방법들이 있습니다:

✅ 1. 컴파일 타임에 타입 확인: typeof (GNU 확장)
c
Copy
Edit
// GCC에서만 사용 가능
#include <stdio.h>

int main() {
    int x = 5;
    typeof(x) y = 10;  // x와 같은 타입 (int)
    printf("%d\n", y);
    return 0;
}
📌 typeof()는 GCC 전용 확장 문법이라서 표준 C에서는 사용할 수 없습니다.

✅ 2. 매크로로 타입 체크 흉내내기 (컴파일 타임)
c
Copy
Edit
#define TYPE_INT(x) _Generic((x), int: 1, default: 0)

int main() {
    int a = 5;
    float b = 3.14;

    if (TYPE_INT(a)) printf("a는 int\n");
    if (TYPE_INT(b)) printf("b는 int\n");
    else printf("b는 int 아님\n");
}
_Generic은 C11 표준에서 도입된 기능

컴파일 타임에 분기

여러 타입을 넣어 조건문처럼 응용 가능

✅ 3. union + enum으로 타입 추적 (런타임용 간접 방법)
c
Copy
Edit
typedef enum { TYPE_INT, TYPE_FLOAT } TypeTag;

typedef struct {
    TypeTag type;
    union {
        int i;
        float f;
    } data;
} Value;

void print_value(Value v) {
    switch (v.type) {
        case TYPE_INT:
            printf("int: %d\n", v.data.i);
            break;
        case TYPE_FLOAT:
            printf("float: %.2f\n", v.data.f);
            break;
    }
}
➡️ 이렇게 하면 런타임에서도 타입 추적 가능
이 방식은 variant, tagged union 구조라고 부릅니다.

✅ 정리
방법	용도	런타임 확인 가능 여부	표준 C 지원
typeof	타입 복사	❌	❌ (GCC 전용)
_Generic	타입 분기	❌ (컴파일 타임)	✅ (C11~)
union + enum	런타임 타입 추적	✅	✅
typeid, typeinfo	❌ C에는 없음 (C++ 전용)	-	-

📌 결론
C에서는 "자료형을 알아내는 것" 자체가 설계 철학에 맞지 않습니다.
대신 enum으로 직접 타입을 관리하거나, 컴파일 타임 분기를 활용해 타입을 구분합니다.

런타임 동적 타입이 필요한 경우는 C가 아닌 C++ 또는 인터프리터 언어(Python 등)가 더 적합합니다.

필요하면 _Generic을 활용한 다형성 매크로도 예제로 보여드릴 수 있어요.
