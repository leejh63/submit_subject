#include <pthread.h>
#include <stdio.h>
#include <unistd.h>

/* 전역 동기화 객체 – 정적 초기화 */
pthread_mutex_t   mtx  = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t    cond = PTHREAD_COND_INITIALIZER;
pthread_once_t    once = PTHREAD_ONCE_INIT;
pthread_key_t     tls_key;

/* once: TLS 소멸자·키 생성 */
void tls_init_once(void)
{
    pthread_key_create(&tls_key, free);   // 스레드 종료 시 free
}

void *worker(void *arg)
{
    /* TLS 변수 준비 */
    pthread_once(&once, tls_init_once);
    int *counter = calloc(1, sizeof(int));
    pthread_setspecific(tls_key, counter);

    /* 임계영역 예시 */
    pthread_mutex_lock(&mtx);
    (*counter)++;
    printf("TID=%lu, local=%d\n", pthread_self(), *counter);
    if (*counter == 3)               /* 마지막 스레드라면 */
        pthread_cond_signal(&cond);  /* 메인에 알림         */
    pthread_mutex_unlock(&mtx);
    return NULL;
}

int main(void)
{
    pthread_t tid[3];
    pthread_attr_t attr;

    /* 속성:  detached 스레드 X, 스택 1 MB 지정 */
    pthread_attr_init(&attr);
    pthread_attr_setstacksize(&attr, 1 << 20);

    for (int i = 0; i < 3; ++i)
        pthread_create(&tid[i], &attr, worker, NULL);

    /* 3개 완료될 때까지 대기 */
    pthread_mutex_lock(&mtx);
    pthread_cond_wait(&cond, &mtx);
    pthread_mutex_unlock(&mtx);

    for (int i = 0; i < 3; ++i)
        pthread_join(tid[i], NULL);

    /* 정리 */
    pthread_key_delete(tls_key);
    pthread_attr_destroy(&attr);
    pthread_mutex_destroy(&mtx);
    pthread_cond_destroy(&cond);
    return 0;
}

#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

pthread_mutex_t lock;
pthread_cond_t cond;
int ready = 0;

void *worker(void *arg) {
    int num = *(int *)arg;
    pthread_mutex_lock(&lock);
    while (!ready)
        pthread_cond_wait(&cond, &lock);
    printf("🔹 Worker %d started!\n", num);
    pthread_mutex_unlock(&lock);
    pthread_exit(NULL);
}

int main(void) {
    pthread_t threads[3];
    pthread_attr_t attr;
    int args[3] = {1, 2, 3};

    // 🔧 속성 설정
    pthread_attr_init(&attr);
    pthread_attr_setstacksize(&attr, 1024 * 1024); // 1MB 스택

    pthread_mutex_init(&lock, NULL);
    pthread_cond_init(&cond, NULL);

    for (int i = 0; i < 3; i++)
        pthread_create(&threads[i], &attr, worker, &args[i]);

    sleep(1); // 일부러 대기

    pthread_mutex_lock(&lock);
    ready = 1;
    pthread_cond_broadcast(&cond); // 모든 스레드 시작
    pthread_mutex_unlock(&lock);

    for (int i = 0; i < 3; i++)
        pthread_join(threads[i], NULL);

    // 🔚 자원 해제
    pthread_mutex_destroy(&lock);
    pthread_cond_destroy(&cond);
    pthread_attr_destroy(&attr);

    return 0;
}

#include <pthread.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

#define NUM_THREADS 3

pthread_mutex_t mutex;
pthread_cond_t cond;
int ready = 0;

void *worker(void *arg) {
    long tid = (long)arg;
    pthread_mutex_lock(&mutex);

    while (!ready) {
        pthread_cond_wait(&cond, &mutex); // ready 될 때까지 대기
    }

    printf("스레드 %ld: 작업 시작\n", tid);
    pthread_mutex_unlock(&mutex);
    return NULL;
}

int main(void) {
    pthread_t threads[NUM_THREADS];
    pthread_mutex_init(&mutex, NULL);
    pthread_cond_init(&cond, NULL);

    for (long i = 0; i < NUM_THREADS; i++) {
        pthread_create(&threads[i], NULL, worker, (void *)i);
    }

    sleep(2); // 작업 준비
    pthread_mutex_lock(&mutex);
    ready = 1;
    pthread_cond_broadcast(&cond); // 모든 스레드 깨움
    pthread_mutex_unlock(&mutex);

    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }

    pthread_mutex_destroy(&mutex);
    pthread_cond_destroy(&cond);

    return 0;
}
