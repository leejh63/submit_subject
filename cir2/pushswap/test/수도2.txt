typedef t_stack s_stack
{
    struct s_stack	*next;
	long long			num;
}	t_stack;

t_stack	*new_num(long long num)
{
	t_stack	*new;

	new = malloc(sizeof(t_stack));
	if (!new)
		return (NULL);
	new -> next = NULL;
	new -> num = num;
	return (new);
}

void	add_f(t_stack **head, t_stack *new)
{
	//널처리
	if (!*head)
	{
		new -> next = NULL;
		*head = new;
		return ;
	}
	new -> next = *head;
	*head = new;
	return ;
}
