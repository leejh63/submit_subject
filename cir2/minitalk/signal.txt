#include <stdio.h>
#include <signal.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>

// 확장 핸들러
void sigusr1_handler(int sig, siginfo_t *info, void *ucontext)
{
    printf("[SIGUSR1] from PID: %d, value: %d\n", info->si_pid, info->si_value.sival_int);
}

// 기본 핸들러
void sigint_handler(int sig)
{
    printf("[SIGINT] Ctrl+C received! Exiting...\n");
    exit(0);
}

int main()
{
    // SIGUSR1: 확장 핸들러 설정
    struct sigaction sa_usr1;
    memset(&sa_usr1, 0, sizeof(sa_usr1));
    sa_usr1.sa_sigaction = sigusr1_handler;
    sa_usr1.sa_flags = SA_SIGINFO; // siginfo_t 사용
    sigemptyset(&sa_usr1.sa_mask);
    sigaction(SIGUSR1, &sa_usr1, NULL);

    // SIGINT: 기본 핸들러 설정 (Ctrl+C)
    signal(SIGINT, sigint_handler);

    printf("PID: %d\n", getpid());
    printf("Send SIGUSR1 with value using `sigqueue` from another terminal.\n");
    printf("Example: sigqueue %d SIGUSR1 <value>\n", getpid());

    // 시그널 올 때까지 대기
    while (1)
        pause();
}

1. “두 시그널이 동시에 들어오면?” – 커널이 처리하는 순서
핵심 포인트	설명
시그널은 ‘동시에’ 실행되지 않는다	커널 안에서는 각 시그널이 비동기적으로 기록(pending) 된다. 같은 순간 발생해도 우선 pending 비트만 세워지고, 사용자 공간으로 돌아가기 전에 한 번에 하나씩 핸들러가 호출된다.
동일 종류는 1 개만 pending	SIGALRM이 3 번 연달아 들어와도 커널 비트는 하나라서 “1 회만” 표시된다. 핸들러가 끝난 뒤에 커널이 다시 같은 시그널을 보낼 수 있다.
우선순서·순서는 구현 의존	리눅스는 보통 오래된 것 먼저(FIFO) 처리를 시도한다. 두 시그널이 정말 같은 지점에서 발생하면 어느 쪽이 먼저 전달될지는 보장되지 않는다.
중첩(재진입) 가능 여부	시그널 A 핸들러 실행 중에 시그널 B가 도착→
 B의 핸들러는 즉시 중첩 실행된다(해당 B가 mask 에 차단돼 있지 않다면).
핸들러 내부에서 커널이 스택 하나 더 얹고 B 핸들러를 호출 → 끝나면 A 핸들러로 복귀.

→ 따라서
SIGALRM과 SIGINT가 “거의 동시에” 오면

커널이 둘 다 pending 상태로 기록

커널이 하나를 골라 먼저 호출 (예: SIGALRM → test_tick)

test_tick 실행 도중 SIGINT가 아직 pending 이고, mask에 없으므로 즉시 test_goodbye가 중첩돼 실행

test_goodbye 리턴 → test_tick 나머지 부분 마저 실행 → 사용자 코드(메인 루프) 복귀.



| **목적** | **시그널 핸들러와 일반 코드(메인 스레드)** 가 **안전하게 공유**할 수 있는 최소 단위 타입을 제공. |
| **보장** | - **하나의 쓰기와 하나의 읽기**가 겹치지 않으면 “원자적(tear-free)으로” 동작한다.<br>- 메모리 폭이 **int 이하**이고, 대부분 플랫폼에서 `int` 나 `volatile int` 와 동일하지만 *반드시* 그렇다고 가정해선 안 됨. |
| **한계** | - 멀티 쓰레드 환경의 “lock-free 원자성”을 보장하지 않는다.<br>- 복수 스레드가 동시에 **쓰기** 하면 여전히 데이터 레이스 → UB. |

> 📌 **핵심**: *시그널 핸들러는 언제든 메인 플로우를 가로질러 들어올 수 있으므로*  
> 순수 C 의 언어 차원에서 **안전한 공유 변수**는 `sig_atomic_t` 단 하나뿐입니다.

## 왜 `volatile sig_atomic_t` 로 선언할까?

1. **`sig_atomic_t`** → 원자적(read/write 찢김 방지).  
2. **`volatile`**   → 컴파일러에게 “이 변수는 **예상치 못한 시점**에 변할 수 있어. 최적화로 캐싱하지 마라.”

조합 예:

```c
#include <signal.h>
#include <stdio.h>
#include <unistd.h>

/* 1) 안전 공유 변수 */
static volatile sig_atomic_t ticked = 0;

void handler(int sig)
{
    (void)sig;
    ticked = 1;        /* 2) write: tear-free */
}

int main(void)
{
    signal(SIGUSR1, handler);

    while (!ticked)    /* 3) read: 매 루프 새로 메모리 로드 */
        pause();       /* 절전 대기 */

    puts("got SIGUSR1!");
    return 0;
}


sig_atomic_t ≠ 진정한 “lock-free atomic type”
→ 시그널 안전을 위한 최소 보장치일 뿐.

여러 스레드가 동시에 쓰거나 복합 연산을 한다면
→ sig_atomic_t 는 무력, _Atomic / mutex 로 넘어가야 합니다.

struct sigaction {
    union {
        void     (*sa_handler)(int);
        void     (*sa_sigaction)(int, siginfo_t *, void *);
    };
    sigset_t    sa_mask;
    int         sa_flags;
    void      (*sa_restorer)(void); // (일반적으로 사용하지 않음, 표준 아님)
};


typedef struct {
    int      si_signo;   // 시그널 번호
    int      si_errno;   // 관련 에러 번호
    int      si_code;    // 시그널 발생 원인 코드
    pid_t    si_pid;     // 시그널을 보낸 프로세스의 PID
    uid_t    si_uid;     // 시그널을 보낸 프로세스의 UID
    int      si_status;  // 종료 상태(자식 프로세스 관련 시그널)
    clock_t  si_utime;   // 사용자 모드에서 사용한 시간
    clock_t  si_stime;   // 커널 모드에서 사용한 시간
    void    *si_addr;    // 잘못된 메모리 접근 주소 등
    union sigval si_value; // 시그널에 전달된 값
    // ... 기타 필드(시그널 종류에 따라 다름)
} siginfo_t;

6. 안전 수칙 & 디버깅 팁
핸들러 안에선 async-signal-safe 함수만 → write, _exit, sig_atomic_t 변수 조작 정도.

멀티스레드라면 전용 “시그널 스레드” 또는 pthread_sigmask()로 다른 스레드에서 block.

상태 추적: sigaction(SIGUSR1, NULL, &cur) / cat /proc/$pid/status | grep Sig

gdb 핸들러 중단 → frame 0, print *info, info signals 로 siginfo_t 내용 확인.

실시간 시그널 범위는 런타임에 sysconf(_SC_RTSIG_SIGQ_MAX) / SIGRTMIN, SIGRTMAX 로 점검.

7. 30초 치트시트
해야 할 일	최소한의 코드/명령
데이터 포함 시그널 보내기	sigqueue(pid, SIGUSR1, (union sigval){.sival_int = x});
3-인자 핸들러 등록	`sa.sa_flags = SA_SIGINFO
핸들러에서 발신자 PID / 값 읽기	info->si_pid, info->si_value.sival_int
잠자는 스레드 깨우기	서버 루프에 pause() or sigsuspend()
“잠시만 다른 핸들러”	sigaction(sig,&tmp,&bak); …; sigaction(sig,&bak,0);

이걸 기억하면 끝 ✔︎
설정은 sigaction + SA_SIGINFO.

전송은 sigqueue (+ 실시간 시그널이면 큐잉).

전달은 “커널→유저 복귀 직전”, 싱글 CPU라면 스케줄러가 한 박자 조율.

oldact 은 백업·조회용, 필요 없으면 NULL.

IPI 는 멀티코어 전용 “다른 코어 깨우기” 하드웨어 인터럽트.

이제 PID 확인, 값 전달, 재진입·재시작 규칙, 내부 타이밍까지 모두 머릿속에 들어갔으니, 42 Minitalk든 실무 서버든 시그널 디버깅이 한결 선명해질 것입니다.
